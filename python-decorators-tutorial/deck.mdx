```py title="A Simple Decorator" subtitle="Let's define an incredibly simple decorator that wraps our function in 2 print statements"
def mutate(method):
    def newmethod(*args, **kwargs):
        print("Executing Method")
        method(*args, **kwargs)
        print("Finished Executing Method")
    return newmethod

@mutate
def myFunction():
    print("Hello")
```

```py title="We can use the functools package" subtitle="This will help retain our programs traceability!"
from functools import *

def mutate(method):
    @wraps(method)
    def newmethod(*args, **kwargs):
        print("Executing Method")
        method(*args, **kwargs)
        print("Finished Executing Method")
    return newmethod

@mutate
def myFunction():
    print("Hello")

myFunction()
```
